001package hw5.test;
002
003import org.junit.BeforeClass;
004import org.junit.Test;
005import hw5.*;
006
007public class GraphEdgeTest {
008        private static GraphEdge g1;
009        private static GraphEdge g2;
010        
011        @BeforeClass 
012    public static void setupBeforeTests() throws Exception {
013        g1 = new GraphEdge(new GraphNode("a"), new GraphNode("c"), "hi");
014        g2 = new GraphEdge(new GraphNode("a"), new GraphNode("b"));
015    }
016        
017        @Test
018        public void testConstructor() {
019                new GraphEdge();
020                new GraphEdge(new GraphNode("a"), new GraphNode("c"));
021                new GraphEdge(new GraphNode("c"), new GraphNode("d"), "HELLO");
022        }
023        
024        @Test
025        public void testGetFields() {
026                assert(g1.getFrom().equals(new GraphNode("a")));
027                assert(g1.getTo().equals(new GraphNode("c")));
028                assert(g1.getFrom().equals(new GraphNode("a")));
029                assert(g1.getTo().equals(new GraphNode("c")));
030                assert(!g1.getFrom().equals(new GraphNode("q")));
031                assert(g1.data().equals("hi"));
032                assert(!g1.data().equals("nono"));
033        }
034        
035        @Test
036        public void testToString() {
037                assert(g1.toString().equals("<a,c>"));
038                assert(g2.toString().equals("<a,b>"));
039        }
040}
