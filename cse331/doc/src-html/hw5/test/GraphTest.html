001package hw5.test;
002
003import java.util.*;
004import org.junit.BeforeClass;
005import org.junit.Test;
006import hw5.*;
007
008public class GraphTest {
009        private static Graph g1;
010        
011        private static List<GraphEdge> x1;
012        private static List<GraphEdge> x2;
013        
014        private static Set<GraphNode> y1;
015        
016        @BeforeClass 
017    public static void setupBeforeTests() throws Exception {
018                
019                x1 = new ArrayList<GraphEdge>();
020                x2 = new ArrayList<GraphEdge>();
021                y1 = new TreeSet<GraphNode>();
022                
023                x1.add(new GraphEdge(new GraphNode("a"), new GraphNode("b")));
024                x1.add(new GraphEdge(new GraphNode("a"), new GraphNode("c")));
025                x1.add(new GraphEdge(new GraphNode("b"), new GraphNode("c")));
026                x1.add(new GraphEdge(new GraphNode("b"), new GraphNode("c")));
027                x1.add(new GraphEdge(new GraphNode("c"), new GraphNode("d")));
028                
029                x2.add(new GraphEdge(new GraphNode("a"), new GraphNode("b")));
030                x2.add(new GraphEdge(new GraphNode("a"), new GraphNode("c")));
031                
032                y1.add(new GraphNode("a")); 
033                y1.add(new GraphNode("b")); 
034                y1.add(new GraphNode("c")); 
035                y1.add(new GraphNode("d"));
036                
037                g1 = new Graph(x1);
038    }
039        
040        @Test
041        public void testConstructor() {
042                new Graph();
043                new Graph(x2);
044        }
045        
046        @Test
047        public void testAllContains() {
048                assert(g1.containsEdge(new GraphEdge(new GraphNode("a"), new GraphNode("b"))));
049                assert(!g1.containsEdge(new GraphEdge(new GraphNode("q"), new GraphNode("z"))));
050                assert(g1.containsNode(new GraphNode("a")));
051                assert(!g1.containsNode(new GraphNode("y")));
052        }
053        
054        @Test
055        public void testConnected() {
056                assert(g1.isConnected(new GraphNode("a"), new GraphNode("b")));
057                assert(!g1.isConnected(new GraphNode("a"), new GraphNode("a")));
058        }
059        
060        @Test
061        public void testEmpty() {
062                assert(new Graph().isEmpty());
063                assert(!g1.isEmpty());
064        }
065        
066        @Test
067        public void testAllGet() {
068                assert(y1.equals(g1.getAllNodes()));
069                assert(x1.equals(g1.getAllEdges()));
070                assert(x2.equals(g1.getAllEdgesFrom(new GraphNode("a"))));
071        }
072        
073        @Test
074        public void testAllRemove() {
075                Graph tempGraph = new Graph(g1.getAllEdges());
076                List<GraphEdge> temp = new ArrayList<GraphEdge>(x1);
077                Set<GraphNode> temp2 = new HashSet<GraphNode>(y1);
078                
079                temp.remove(new GraphEdge(new GraphNode("a"), new GraphNode("b")));
080                temp.remove(new GraphEdge(new GraphNode("a"), new GraphNode("c")));
081                temp2.remove(new GraphNode("a"));
082                tempGraph.removeNode(new GraphNode("a"));
083                assert(temp.equals(tempGraph.getAllEdges()));
084                assert(temp2.equals(tempGraph.getAllNodes()));
085                
086                tempGraph.removeEdge(new GraphEdge(new GraphNode("c"), new GraphNode("d")));
087                temp.remove(new GraphEdge(new GraphNode("c"), new GraphNode("d")));
088                assert(temp.equals(tempGraph.getAllEdges()));
089                
090                tempGraph.removeAllEdge(new GraphEdge(new GraphNode("b"), new GraphNode("c")));
091                temp.remove(new GraphEdge(new GraphNode("b"), new GraphNode("c")));
092                temp.remove(new GraphEdge(new GraphNode("b"), new GraphNode("c")));
093                assert(temp.equals(tempGraph.getAllEdges()));
094        }
095        
096        @Test
097        public void testAllAdd() {
098                Graph tempGraph = new Graph(g1.getAllEdges());
099                List<GraphEdge> temp = new ArrayList<GraphEdge>(x1);
100                Set<GraphNode> temp2 = new HashSet<GraphNode>(y1);
101                
102                tempGraph.addNode(new GraphNode("z"));
103                temp2.add(new GraphNode("z"));
104                assert(temp2.equals(tempGraph.getAllNodes()));
105                
106                tempGraph.addEdge(new GraphEdge(new GraphNode("z"), new GraphNode("a")));
107                temp.add(new GraphEdge(new GraphNode("z"), new GraphNode("a")));
108                assert(temp.equals(tempGraph.getAllEdges()));
109        }
110}
