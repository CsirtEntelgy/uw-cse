003import java.io.*;

005import hw5.*;
006
007/**
008 * This class implements a testing driver which reads test scripts
009 * from files for testing Graph.
010 **/
011public class HW5TestDriver {
012
013    public static void main(String args[]) {
014        try {
015            if (args.length > 1) {
016                printUsage();
017                return;
018            }
019
020            HW5TestDriver td;
021
022            if (args.length == 0) {
023                td = new HW5TestDriver(new InputStreamReader(System.in),
024                                       new OutputStreamWriter(System.out));
025            } else {
026
027                String fileName = args[0];
028                File tests = new File (fileName);
029
030                if (tests.exists() || tests.canRead()) {
031                    td = new HW5TestDriver(new FileReader(tests),
032                                           new OutputStreamWriter(System.out));
033                } else {
034                    System.err.println("Cannot read from " + tests.toString());
035                    printUsage();
036                    return;
037                }
038            }
039
040            td.runTests();
041
042        } catch (IOException e) {
043            System.err.println(e.toString());
044            e.printStackTrace(System.err);
045        }
046    }
047
048    private static void printUsage() {
049        System.err.println("Usage:");
050        System.err.println("to read from a file: java hw5.test.HW5TestDriver <name of input script>");
051        System.err.println("to read from standard in: java hw5.test.HW5TestDriver");
052    }
053
054    /** String -> Graph: maps the names of graphs to the actual graph **/
055    private final Map<String, Graph> graphs = new HashMap<String, Graph>();
056    private final PrintWriter output;
057    private final BufferedReader input;
058
059    /**
060     * @requires r != null && w != null
061     *
062     * @effects Creates a new HW5TestDriver which reads command from
063     * <tt>r</tt> and writes results to <tt>w</tt>.
064     **/
065    public HW5TestDriver(Reader r, Writer w) {
066        input = new BufferedReader(r);
067        output = new PrintWriter(w);
068    }
069
070    /**
071     * @effects Executes the commands read from the input and writes results to the output
072     * @throws IOException if the input or output sources encounter an IOException
073     **/
074    public void runTests()
075        throws IOException
076    {
077        String inputLine;
078        while ((inputLine = input.readLine()) != null) {
079            if ((inputLine.trim().length() == 0) ||
080                (inputLine.charAt(0) == '#')) {
081                // echo blank and comment lines
082                output.println(inputLine);
083            }
084            else
085            {
086                // separate the input line on white space
087                StringTokenizer st = new StringTokenizer(inputLine);
088                if (st.hasMoreTokens()) {
089                    String command = st.nextToken();
090
091                    List<String> arguments = new ArrayList<String>();
092                    while (st.hasMoreTokens()) {
093                        arguments.add(st.nextToken());
094                    }
095
096                    executeCommand(command, arguments);
097                }
098            }
099            output.flush();
100        }
101    }
102
103    private void executeCommand(String command, List<String> arguments) {
104        try {
105            if (command.equals("CreateGraph")) {
106                createGraph(arguments);
107            } else if (command.equals("AddNode")) {
108                addNode(arguments);
109            } else if (command.equals("AddEdge")) {
110                addEdge(arguments);
111            } else if (command.equals("ListNodes")) {
112                listNodes(arguments);
113            } else if (command.equals("ListChildren")) {
114                listChildren(arguments);
115            } else {
116                output.println("Unrecognized command: " + command);
117            }
118        } catch (Exception e) {
119            output.println("Exception: " + e.toString());
120        }
121    }
122
123    private void createGraph(List<String> arguments) {
124        if (arguments.size() != 1) {
125            throw new CommandException("Bad arguments to CreateGraph: " + arguments);
126        }
127
128        String graphName = arguments.get(0);
129        createGraph(graphName);
130    }
131
132    private void createGraph(String graphName) {
133        if(!graphs.containsKey(graphName)) {
134                graphs.put(graphName, new Graph());
135                output.println("created graph " + graphName);
136        }
137    }
138
139    private void addNode(List<String> arguments) {
140        if (arguments.size() != 2) {
141            throw new CommandException("Bad arguments to addNode: " + arguments);
142        }
143
144        String graphName = arguments.get(0);
145        String nodeName = arguments.get(1);
146
147        addNode(graphName, nodeName);
148    }
149
150    private void addNode(String graphName, String nodeName) {
151        if(!graphs.get(graphName).containsNode(new GraphNode(nodeName))) {
152                graphs.get(graphName).addNode(new GraphNode(nodeName));;
153                output.println("added node " + nodeName + " to " + graphName);
154        }
155    }
156
157    private void addEdge(List<String> arguments) {
158        if (arguments.size() != 4) {
159            throw new CommandException("Bad arguments to addEdge: " + arguments);
160        }
161
162        String graphName = arguments.get(0);
163        String parentName = arguments.get(1);
164        String childName = arguments.get(2);
165        String edgeLabel = arguments.get(3);
166
167        addEdge(graphName, parentName, childName, edgeLabel);
168    }
169
170    private void addEdge(String graphName, String parentName, String childName,
171            String edgeLabel) {
172        graphs.get(graphName).addEdge(new GraphEdge(new GraphNode(parentName), new GraphNode(childName), edgeLabel));
173        output.println("added edge " + edgeLabel + " from " + parentName + " to " + childName + " in " + graphName);
174    }
175
176    private void listNodes(List<String> arguments) {
177        if (arguments.size() != 1) {
178            throw new CommandException("Bad arguments to listNodes: " + arguments);
179        }
180
181        String graphName = arguments.get(0);
182        listNodes(graphName);
183    }
184
185    private void listNodes(String graphName) {
186        output.println(graphName + " contains:");
187        output.println(graphs.get(graphName).getAllNodes());
188    }
189
190    private void listChildren(List<String> arguments) {
191        if (arguments.size() != 2) {
192            throw new CommandException("Bad arguments to listChildren: " + arguments);
193        }
194
195        String graphName = arguments.get(0);
196        String parentName = arguments.get(1);
197        listChildren(graphName, parentName);
198    }
199
200    private void listChildren(String graphName, String parentName) {
201        // Insert your code here.

203        output.println(graphs.get(graphName).getAllEdgesFrom(new GraphNode(parentName)));
204        // ___ = graphs.get(graphName);
205        // output.println(...);
206    }
207
208    /**
209     * This exception results when the input file cannot be parsed properly
210     **/
211    static class CommandException extends RuntimeException {
212
213        public CommandException() {
214            super();
215        }
216        public CommandException(String s) {
217            super(s);
218        }
219
220        public static final long serialVersionUID = 3495;
221    }
