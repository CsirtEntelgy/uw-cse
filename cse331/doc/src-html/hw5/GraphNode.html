003/**
004 * GraphNode is an immutable class for representing a node.
005 * Immutable meaning the user cannot change the data it holds once it is set by the constructor.
006 * <p>
007 *
008 * A GraphNode can be described as a single String
009 * where each GraphNode are characterized by the string it holds.
010 * <p>
011 *
012 * An example of a GraphEdge includes: GraphNode("a")
013 *                                                                         GraphNode("Hello")
014 *
015 * @author Young Bin Cho
016 */
017
018public class GraphNode implements Comparable<GraphNode>{
019        
020        // This class represents a node, or a point
021        // Invariant: All fields must not be null
022        
023        /*The data this node holds*/
024        private final String data;
025        
026        /**
027         * @effects constructs an empty GraphNode
028         */
029        public GraphNode() {
030                data = "";
031                checkRep();
032        }
033        
034        /**
035         * constructs a GraphNode with given string
036         * @requires data != null
037         * @param data string to be set as the data
038         */
039        public GraphNode(String data) {
040                this.data = data;
041                checkRep();
042        }
043        
044        /**
045         * constructs a GraphNode with given GraphNode
046         * @requires g != null
047         * @param g GraphNode to set as this one
048         */
049        public GraphNode(GraphNode g) {
050                this.data = g.data;
051        }
052        
053        /**
054         * returns the data field
055         * @return copy of data
056         */
057        public String data() {
058                return this.data;
059        }
060        
061        /**
062         * below are auto-generated methods for the override of equals
063         */
064        @Override
065        public int hashCode() {
066                final int prime = 31;
067                int result = 1;
068                result = prime * result + ((data == null) ? 0 : data.hashCode());
069                return result;
070        }
071        @Override
072        public boolean equals(Object obj) {
073                if (this == obj)
074                        return true;
075                if (obj == null)
076                        return false;
077                if (getClass() != obj.getClass())
078                        return false;
079                GraphNode other = (GraphNode) obj;
080                if (data == null) {
081                        if (other.data != null)
082                                return false;
083                } else if (!data.equals(other.data))
084                        return false;
085                return true;
086        }
087        
088        /**
089         * returns a concatenated string for a node
090         * @requires data != null
091         * @return string form of node
092         */
093        @Override
094        public String toString() {
095                return this.data;
096        }
097        
098        /**
099         * returns a integer value for comparing an object to this node
100         * @requires arg != null
101         * @return negative, if smaller, 0 if equal, and positive if bigger
102         */
103        @Override
104        public int compareTo(GraphNode arg) {
105                return data.compareTo(arg.data);
106        }
107        
108        /**
109         * Checks that the representation invariant holds (if any)
110         */
111        private void checkRep() {
112                assert(data != null);
113        }
114}
