003/**
004 * GraphEdge is an immutable class for representing a line between two GraphNodes.
005 * GraphEdge is a pair of GraphNodes with a data field.
006 * Immutable meaning the user cannot change the data it holds once it is set by the constructor.
007 * <p>
008 *
009 * A GraphEdge can be described as, 2 GraphNodes, from and to, and a String that holds its data
010 * where from is where the edge starts and to is where the edge ends
011 * String is the data this edge holds.
012 * <p>
013 *
014 * An example of a GraphEdge includes: GraphEdge(GraphNode("a"), GraphNode("b"))
015 *                                                                         GraphEdge(GraphNode("b"), GraphNode("c"), "path from b to c")
016 *
017 * @author Young Bin Cho
018 */
019
020public class GraphEdge implements Comparable<GraphEdge>{
021        
022        // This class represents a edge from one GraphNode to another
023        // Invariant: All fields must not be null
024
025        /*private fields that hold the data for this class*/
026        private final GraphNode from;
027        private final GraphNode to;
028        private final String data;
029        
030        /**
031         * @effects constructs an empty GraphEdge
032         */
033        public GraphEdge() {
034                from = new GraphNode();
035                to = new GraphNode();
036                data = "";
037                checkRep();
038        }
039        
040        /**
041         * constructor that takes only the GraphNodes from and to
042         * @param from GraphNode where the edge starts
043         * @param to GraphNode where the edge ends
044         * @modifies this
045         * @effects constructs new GraphEdge out of two GraphNodes
046         */
047        public GraphEdge(GraphNode from, GraphNode to) {
048                this(from, to, "");
049        }
050        
051        /**
052         * constructor that takes two GraphNodes and a string
053         * @requires from != null && to != null && data != null
054         * @param from GraphNode where the edge starts
055         * @param to GraphNode where the edge ends
056         * @param data string this edge holds
057         * @modifies this
058         * @effects constructs new GraphEdge from two GraphNodes and a string
059         */
060        public GraphEdge(GraphNode from, GraphNode to, String data) {
061                this.from = from;
062                this.to = to;
063                this.data = data;
064                checkRep();
065        }
066        
067        /**
068         * returns the from field
069         * @return copy of GraphNode from
070         */
071        public GraphNode getFrom() {
072                return new GraphNode(from);
073        }
074        
075        /**
076         * returns the to field
077         * @return copy of GraphNode to
078         */
079        public GraphNode getTo() {
080                return new GraphNode(to);
081        }
082        
083        /**
084         * returns the data field
085         * @return copy of String data
086         */
087        public String data() {
088                return this.data;
089        }
090
091        /**
092         * below are auto-generated methods for the override of equals
093         */
094        @Override
095        public int hashCode() {
096                final int prime = 31;
097                int result = 1;
098                result = prime * result + ((data == null) ? 0 : data.hashCode());
099                result = prime * result + ((from == null) ? 0 : from.hashCode());
100                result = prime * result + ((to == null) ? 0 : to.hashCode());
101                return result;
102        }
103        @Override
104        public boolean equals(Object obj) {
105                if (this == obj)
106                        return true;
107                if (obj == null)
108                        return false;
109                if (getClass() != obj.getClass())
110                        return false;
111                GraphEdge other = (GraphEdge) obj;
112                if (data == null) {
113                        if (other.data != null)
114                                return false;
115                } else if (!data.equals(other.data))
116                        return false;
117                if (from == null) {
118                        if (other.from != null)
119                                return false;
120                } else if (!from.equals(other.from))
121                        return false;
122                if (to == null) {
123                        if (other.to != null)
124                                return false;
125                } else if (!to.equals(other.to))
126                        return false;
127                return true;
128        }
129        
130        /**
131         * returns a concatenated string for a edge
132         * @requires from != null && to != null
133         * @return string form of edge
134         */
135        @Override
136        public String toString() {
137                return "<"+from.data()+","+to.data()+">";
138        }
139        
140        /**
141         * returns a integer value for comparing an object to this edge
142         * @requires arg != null
143         * @return negative, if smaller, 0 if equal, and positive if bigger
144         */
145        @Override
146        public int compareTo(GraphEdge arg) {
147                return toString().compareTo(arg.toString());
148        }
149        
150        /**
151         * Checks that the representation invariant holds (if any)
152         */
153        private void checkRep() {
154                assert(from != null && to != null && data != null);
155        }
156}
